# WSL helper to serve Stickmotion Studio
# If you prefer a one-liner, just run from anywhere:
#   ./run -stickman --port 8006

# Manual steps (equivalent to the script) if you want to copy/paste:
sudo mkdir -p <DRIVE_MOUNT>
sudo mount -t drvfs <DRIVE_LETTER>: <DRIVE_MOUNT> || true
cd "<PROJECTS_DIR>/Stickman-animation-app"
python3 -m http.server 8006 --bind 0.0.0.0
